def log(msg: str, level: str = "INFO", store: str = "pipeline", container=None):
    """
    store='pipeline' → pipeline log
    store='scraper'  → scraper log
    """
    ts = datetime.datetime.now().strftime("%H:%M:%S.%f")[:-3]
    entry = {"ts": ts, "level": level, "msg": str(msg)}
    if store == "scraper":
        st.session_state.scrape_logs.append(entry)
    else:
        st.session_state.log_lines.append(entry)
    
    # If a container is provided, update the UI immediately
    if container:
        render_log(container, store=store)


def slog(msg: str, level: str = "INFO", container=None):
    """Shorthand for scraper log."""
    log(msg, level, store="scraper", container=container)


def render_log(container, store: str = "pipeline", n: int = 300):
    if store == "scraper":
        lines = st.session_state.scrape_logs[-n:]
    else:
        lines = st.session_state.log_lines[-n:]
    rows = [
        f'<div><span class="log-ts">{e["ts"]}</span>'
        f'<span class="log-{e["level"]}">[{e["level"]}]</span> {e["msg"]}</div>'
        for e in lines
    ]
    container.markdown(f'<div class="log-box">{"".join(rows)}</div>', unsafe_allow_html=True)

def sanitize_foldername(name: str) -> str:
    import html
    name = html.unescape(name)
    safe = re.sub(r'[\\/*?:"<>|]', "", name)
    # Remove emojis and non-ascii for even safer filenames if needed, 
    # but let's keep it simple first and just ensure basic safety.
    return safe.strip(". ")[:80]

def get_video_duration(video_path: str) -> float:
    try:
        r = subprocess.run([
            "ffprobe", "-v", "error", "-show_entries", "format=duration",
            "-of", "default=noprint_wrappers=1:nokey=1", video_path
        ], capture_output=True, text=True, check=True, timeout=15, encoding="utf-8", errors="replace")
        return float(r.stdout.strip())
    except Exception:
        return 30.0


def is_video_post(post: dict) -> bool:
    """Strictly identify if a Reddit post contains a downloadable video."""
    if post.get("is_video"):
        return True
    if post.get("post_hint") in ["hosted:video", "rich:video"]:
        return True

    # Check domain and URL for known video hosting sites
    url = post.get("url", "").lower()
    video_domains = [
        "v.redd.it", "youtube.com", "youtu.be", "streamable.com",
        "gfycat.com", "tiktok.com", "vimeo.com"
    ]
    if any(d in url for d in video_domains):
        return True

    # Check crosspost parent data
    if "crosspost_parent_list" in post and post["crosspost_parent_list"]:
        for cp in post["crosspost_parent_list"]:
            if cp.get("is_video") or cp.get("post_hint") in ["hosted:video", "rich:video"]:
                return True
            cp_url = cp.get("url", "").lower()
            if any(d in cp_url for d in video_domains):
                return True

    return False


def reddit_get(url: str, params: dict = None) -> Optional[dict]:
    if not url.endswith(".json"):
        url = url.rstrip("/") + ".json"
    try:
        log(f"[HTTP] GET {url[:80]}")
        r = SESSION.get(url, params=params, timeout=15)
        if r.status_code == 429:
            log("[HTTP] Rate limited — waiting 20s", "WARN")
            time.sleep(20)
            r = SESSION.get(url, params=params, timeout=15)
        r.raise_for_status()
        return r.json()
    except Exception as e:
        log(f"[HTTP] Failed: {e}", "ERROR")
        return None

def extract_frames_for_scraper(video_path: str, output_folder: str, video_label: str, n: int = NUM_FRAMES) -> list:
    """Extract N evenly-spaced frames from a video (scraper version with scraper logs)."""
    duration = get_video_duration(video_path)
    frames_folder = os.path.join(output_folder, "frames")
    os.makedirs(frames_folder, exist_ok=True)

    start = duration * 0.05
    end   = duration * 0.95
    timestamps = [start + (end - start) * i / max(1, n - 1) for i in range(n)]

    slog(f"🎞 Extracting {n} frames from {video_label} (dur={duration:.1f}s, {start:.1f}–{end:.1f}s)", "FRAME")

    frame_paths = []
    for i, ts in enumerate(timestamps):
        path = os.path.join(frames_folder, f"frame_{i+1:02d}.jpg")
        try:
            subprocess.run([
                "ffmpeg", "-ss", str(ts), "-i", video_path,
                "-frames:v", "1", "-q:v", "2", path, "-y"
            ], check=True, capture_output=True, timeout=20)
            frame_paths.append(path)
            slog(f"   📸 Frame {i+1:02d}/{n}: t={ts:.1f}s → {os.path.basename(path)}", "FRAME")
        except Exception as e:
            slog(f"   ❌ Frame {i+1:02d} failed: {e}", "ERROR")

    slog(f"✅ Extracted {len(frame_paths)}/{n} frames for {video_label}", "SUCCESS")
    return frame_paths


def scrape_youtube(query: str, limit: int, db: dict, live_container=None) -> list:
    slog("═" * 50, "PLATFORM", container=live_container)
    slog(f"▶ YOUTUBE — '{query}' (limit={limit})", "PLATFORM", container=live_container)
    slog("═" * 50, "PLATFORM", container=live_container)

    if not YOUTUBE_AVAILABLE:
        slog("❌ google-api-python-client not installed. pip install google-api-python-client", "ERROR", container=live_container)
        return []

    try:
        youtube = yt_build("youtube", "v3", developerKey=YOUTUBE_API_KEY)
        slog("YouTube API client ready", "INFO", container=live_container)

        response = youtube.search().list(
            q=query, part="snippet", type="video",
            maxResults=limit, order="relevance"
        ).execute()

        items = response.get("items", [])
        slog(f"Found {len(items)} video(s)", "SUCCESS", container=live_container)
        if not items:
            slog("No results for this query.", "WARN", container=live_container)
            return []

        results = []
        for idx, item in enumerate(items, 1):
            vid   = item["id"]["videoId"]
            title = item["snippet"]["title"]
            url   = f"https://www.youtube.com/watch?v={vid}"
            safe  = sanitize_foldername(title) or f"yt_{vid}"

            slog(f"──── [{idx}/{len(items)}] {title[:60]}", "INFO", container=live_container)
            slog(f"🔗 {url}", "INFO", container=live_container)

            folder_path = os.path.join(YOUTUBE_DIR, safe)
            os.makedirs(folder_path, exist_ok=True)
            video_path  = os.path.join(folder_path, f"{vid}.mp4")

            slog(f"   ⬇ [1/4] Downloading…", "DOWNLOAD", container=live_container)
            try:
                # Fixed format for better compatibility
                res = subprocess.run([
                    "yt-dlp", "--no-warnings", "--quiet",
                    "--extractor-args", "youtube:player_client=android,web_creator",
                    "-f", "bestvideo[ext=mp4]+bestaudio[ext=m4a]/best[ext=mp4]/best",
                    "-o", video_path, "--no-playlist", url
                ], check=True, timeout=120, capture_output=True, text=True, encoding="utf-8", errors="replace")

                if not os.path.exists(video_path):
                    slog(f"   ❌ [1/4] File not found after download for {vid}", "ERROR", container=live_container)
                    if res.stderr: slog(f"      DEBUG: {res.stderr.strip()[:100]}", "WARN", container=live_container)
                    continue

                size_kb = os.path.getsize(video_path) // 1024
                slog(f"   ✅ [1/4] Downloaded: {size_kb} KB", "SUCCESS", container=live_container)

                slog(f"   🎞 [2/4] Extracting {NUM_FRAMES} frames…", "FRAME", container=live_container)
                frames = extract_frames_for_scraper(video_path, folder_path, f"YT:{vid}")

                slog(f"   🔍 [3/4] Running Piracy Scan (pHash + PDQ + Audio)…", "INFO", container=live_container)
                # Modified scan_scraped_video to support live logs internally if needed
                scan_res = scan_scraped_video_live(video_path, title, url, "youtube", db, live_container)

                slog(f"   💾 [4/4] Saving Metadata…", "INFO", container=live_container)
                meta = {
                    "platform": "youtube", "video_id": vid, "title": title,
                    "url": url, "folder": folder_path,
                    "scraped_at": datetime.datetime.now().isoformat(),
                    "frames_count": len(frames),
                    "scan_result": scan_res
                }
                with open(os.path.join(folder_path, "meta.json"), "w") as f:
                    json.dump(meta, f, indent=2)

                results.append({**meta, "video_path": video_path, "frames": frames})

            except subprocess.CalledProcessError as e:
                err_info = e.stderr.strip() if e.stderr else str(e)
                slog(f"   ❌ Download failed: {err_info[:200]}", "ERROR", container=live_container)
            except subprocess.TimeoutExpired:
                slog(f"   ⏰ Timeout for {vid}", "ERROR", container=live_container)
            except Exception as e:
                slog(f"   ❌ Error: {e}", "ERROR", container=live_container)

        slog(f"▶ YOUTUBE DONE — {len(results)}/{len(items)} processed", "PLATFORM", container=live_container)
        return results

    except Exception as e:
        slog(f"❌ YouTube API error: {e}", "ERROR", container=live_container)
        return []

def scrape_reddit(query: str, limit: int, db: dict, live_container=None) -> list:
    slog("═" * 50, "PLATFORM", container=live_container)
    slog(f"▶ REDDIT — '{query}' (limit={limit})", "PLATFORM", container=live_container)
    slog("═" * 50, "PLATFORM", container=live_container)

    video_posts = []
    after = None
    max_pages = 10 

    while len(video_posts) < limit and max_pages > 0:
        max_pages -= 1
        params = {"q": query, "sort": "relevance", "limit": 100, "raw_json": 1}
        if after:
            params["after"] = after

        try:
            data = SESSION.get(
                "https://www.reddit.com/search.json",
                params=params,
                timeout=15
            )
            if data.status_code == 429:
                slog("⚠ Reddit rate-limited — waiting 20s", "WARN", container=live_container)
                time.sleep(20)

            break

        children = res_json.get("data", {}).get("children", [])

            if is_video_post(p):
                if not any(v.get("id") == p.get("id") for v in video_posts):
                    video_posts.append(p)

                if len(video_posts) >= limit:
                    break

        slog(f"Fetched {len(posts)} posts, found {found_count} new videos. Total: {len(video_posts)}/{limit}", "INFO", container=live_container)
        
        after = res_json.get("data", {}).get("after")

        return []

    slog(f"✅ Finalizing {len(video_posts)} video posts", "SUCCESS", container=live_container)

        downloaded = False

            if not attempt_url or downloaded:
                continue

                res = subprocess.run([
                    "yt-dlp", "--no-warnings", "--quiet",
                    "--extractor-args", "youtube:player_client=android,web_creator",
                    "-f", "bestvideo[ext=mp4]+bestaudio[ext=m4a]/best[ext=mp4]/best",
                    "-o", video_path, "--no-playlist", attempt_url
                ], capture_output=True, text=True, timeout=120, encoding="utf-8", errors="replace")
                
                if os.path.exists(video_path):
                    downloaded = True
                    break
            except Exception:
                pass

        if downloaded:
            size_kb = os.path.getsize(video_path) // 1024
            slog(f"   ✅ [1/4] Downloaded: {size_kb} KB", "SUCCESS", container=live_container)

# ── Agent 1: Ingestion ──────────────────

def ingest_video(url: str, output_path: str, retries: int = 2) -> bool:
    for attempt in range(1, retries + 1):
        log(f"[INGEST] Attempt {attempt}: {url[:70]}")
        try:
            r = subprocess.run([
                "yt-dlp", "--quiet", "--no-warnings",
                "--merge-output-format", "mp4",
                "--no-playlist", "-o", output_path, url
            ], capture_output=True, timeout=90)
            if os.path.exists(output_path):
                size = os.path.getsize(output_path)
                log(f"[INGEST] Downloaded {size//1024} KB → {os.path.basename(output_path)}", "SUCCESS")
                return True
            stderr = r.stderr.decode("utf-8", errors="replace")[:150]
            log(f"[INGEST] yt-dlp stderr: {stderr}", "WARN")
        except subprocess.TimeoutExpired:
            log(f"[INGEST] Timeout on attempt {attempt}", "WARN")
        except Exception as e:
            log(f"[INGEST] Exception: {e}", "ERROR")
        time.sleep(1.5)
    return False


def ingest_multiple(urls: list, out_dir: str) -> dict:
    """Ingest multiple URLs concurrently. Returns {url: local_path}."""
    results = {}
    os.makedirs(out_dir, exist_ok=True)

    def _dl(url):
        uid  = hashlib.md5(url.encode()).hexdigest()[:8]
        path = os.path.join(out_dir, f"ingest_{uid}.mp4")
        ok   = ingest_video(url, path)
        return (url, path) if ok else None

    log(f"[INGEST] Parallel ingestion of {len(urls)} URLs")
    with concurrent.futures.ThreadPoolExecutor(max_workers=MAX_WORKERS) as pool:
        for result in pool.map(_dl, urls):
            if result:
                results[result[0]] = result[1]
    log(f"[INGEST] Completed: {len(results)}/{len(urls)} successful", "SUCCESS")
    return results


# ── Agent 2: Fingerprinting ─────────────

def _extract_frame_worker(args) -> Optional[str]:
    video_path, ts, out_path = args
    try:
        subprocess.run(
            ["ffmpeg", "-ss", str(ts), "-i", video_path,
             "-frames:v", "1", "-q:v", "2", out_path, "-y"],
            capture_output=True, timeout=20
        )
        return out_path if os.path.exists(out_path) else None
    except Exception:
        return None


def extract_frames_concurrent(video_path: str, label: str, n: int = PIPE_FRAMES) -> list:
    """Extract N frames concurrently for pipeline fingerprinting."""
    dur   = get_video_duration(video_path)
    start = dur * 0.05
    end   = dur * 0.95
    tss   = [start + (end - start) * i / max(1, n - 1) for i in range(n)]

    uid     = hashlib.md5(label.encode()).hexdigest()[:8]
    out_dir = os.path.join(FRAMES_DIR, uid)
    os.makedirs(out_dir, exist_ok=True)

    tasks = [(video_path, ts, os.path.join(out_dir, f"f{i:02d}.jpg")) for i, ts in enumerate(tss)]

    paths = []
    with concurrent.futures.ThreadPoolExecutor(max_workers=MAX_WORKERS) as pool:
        for r in pool.map(_extract_frame_worker, tasks):
            if r:
                paths.append(r)

    log(f"[FINGERPRINT] Extracted {len(paths)}/{n} frames from {os.path.basename(video_path)}")
    return paths


def compute_phash_list(frame_paths: list) -> list:
    hashes = []
    for fp in frame_paths:
        try:
            img = Image.open(fp).convert("RGB")
            hashes.append(str(imagehash.phash(img)))
        except Exception as e:
            log(f"[FINGERPRINT] pHash failed: {e}", "WARN")
    log(f"[FINGERPRINT] Computed {len(hashes)} pHashes")
    return hashes


def compute_pdq_list(frame_paths: list) -> list:
    if not PDQ_AVAILABLE:
        return []
    hashes = []
    for fp in frame_paths:
        try:
            image = cv2.imread(fp)
            if image is None: continue
            image_rgb = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
            hash_vector, quality = pdqhash.compute(image_rgb)
            # Convert bit vector to hex string
            hash_hex = "".join([str(bit) for bit in hash_vector])
            hash_hex = hex(int(hash_hex, 2))[2:].zfill(64)
            hashes.append(hash_hex)
        except Exception as e:
            log(f"[FINGERPRINT] PDQ failed: {e}", "WARN")
    log(f"[FINGERPRINT] Computed {len(hashes)} PDQ hashes")
    return hashes


def compute_audio_fingerprint(video_path: str) -> Optional[str]:
    """Chromaprint / fpcalc audio fingerprint. Gracefully skips if fpcalc not installed."""
    try:
        r = subprocess.run(
            ["fpcalc", "-raw", "-length", "60", video_path],
            capture_output=True, text=True, timeout=30, encoding="utf-8", errors="replace"
        )
        for line in r.stdout.splitlines():
            if line.startswith("FINGERPRINT="):
                fp = line.split("=", 1)[1]
                log(f"[FINGERPRINT] Audio fingerprint: {fp[:30]}…")
                return fp
    except FileNotFoundError:
        log("[FINGERPRINT] fpcalc not found — audio fingerprinting disabled", "WARN")
    except Exception as e:
        log(f"[FINGERPRINT] Audio fingerprint failed: {e}", "WARN")
    return None


def fingerprint_video(video_path: str, label: str) -> dict:
    log(f"[FINGERPRINT] Starting: {os.path.basename(video_path)}")
    frames   = extract_frames_concurrent(video_path, label)
    phashes  = compute_phash_list(frames)
    pdq_hashes = compute_pdq_list(frames)
    audio_fp = compute_audio_fingerprint(video_path)
    return {
        "label":             label,
        "frames":            frames,
        "phashes":           phashes,
        "pdq_hashes":        pdq_hashes,
        "audio_fp":          audio_fp,
        "fingerprinted_at":  datetime.datetime.now().isoformat()
    }


# ── Agent 3: Detection ───────────────────

def phash_similarity(hashes_a: list, hashes_b: list) -> float:
    if not hashes_a or not hashes_b:
        return 0.0
    distances = [imagehash.hex_to_hash(a) - imagehash.hex_to_hash(b)
                 for a in hashes_a for b in hashes_b]
    min_d = min(distances)
    score = max(0.0, 1.0 - min_d / 64.0)
    log(f"[DETECT] pHash: min_dist={min_d} → score={score:.3f}")
    return score


def pdq_similarity(hashes_a: list, hashes_b: list) -> float:
    if not hashes_a or not hashes_b:
        return 0.0
    
    # Calculate min Hamming distance
    min_dist = 256
    for ha in hashes_a:
        v_a = int(ha, 16)
        for hb in hashes_b:
            v_b = int(hb, 16)
            # Efficient Hamming distance via XOR and bit count
            dist = bin(v_a ^ v_b).count('1')
            if dist < min_dist:
                min_dist = dist
    
    # Normalized score: PDQ match is usually < 30 dist. 
    # Using 64 as a threshold for 0.0 score.
    score = max(0.0, 1.0 - (min_dist / 64.0))
    log(f"[DETECT] PDQ: min_dist={min_dist} → score={score:.3f}")
    return score


def audio_similarity(fp_a: Optional[str], fp_b: Optional[str]) -> float:
    if not fp_a or not fp_b:
        log("[DETECT] Audio: fingerprint missing — score=0.0", "WARN")
        return 0.0
    try:
        a = [int(x) for x in fp_a.split(",")]
        b = [int(x) for x in fp_b.split(",")]
        length = min(len(a), len(b), 100)
        match  = sum(32 - bin((x ^ y) & 0xFFFFFFFF).count("1")
                     for x, y in zip(a[:length], b[:length]))
        score  = match / (length * 32)
        log(f"[DETECT] Audio: chromaprint similarity={score:.3f}")
        return score
    except Exception as e:
        log(f"[DETECT] Audio similarity error: {e}", "WARN")
        return 0.0


def detect(suspect_fp: dict, ref_fp: dict) -> dict:
    log(f"[DETECT] Comparing '{suspect_fp['label']}' vs '{ref_fp['label']}'")
    return {
        "phash_score":  phash_similarity(suspect_fp["phashes"],   ref_fp.get("phashes", [])),
        "pdq_score":    pdq_similarity(suspect_fp.get("pdq_hashes", []), ref_fp.get("pdq_hashes", [])),
        "audio_score":  audio_similarity(suspect_fp["audio_fp"],  ref_fp.get("audio_fp")),
    }


# ── Agent 4: Scoring ─────────────────────

def score(signals: dict) -> dict:
    p   = signals.get("phash_score", 0.0)
    pdq = signals.get("pdq_score",   0.0)
    a   = signals.get("audio_score", 0.0)

    has_audio = a > 0
    total_w   = W_PHASH + W_PDQ

    if has_audio:
        total_w += W_AUDIO
        final = (W_PHASH / total_w) * p + (W_PDQ / total_w) * pdq + (W_AUDIO / total_w) * a
    else:
        final = (W_PHASH / total_w) * p + (W_PDQ / total_w) * pdq

    if   final >= THRESHOLD_FLAG:   verdict, level = "FLAG",   "MATCH"
    elif final >= THRESHOLD_REVIEW: verdict, level = "REVIEW", "WARN"
    else:                           verdict, level = "DROP",   "INFO"

    log(f"[SCORE] pHash={p:.2f} PDQ={pdq:.2f} audio={a:.2f} → final={final:.3f} → {verdict}", level)
    return {
        "phash_score":  round(p, 4),
        "pdq_score":    round(pdq, 4),
        "audio_score":  round(a, 4),
        "final_score":  round(final, 4),
        "verdict":      verdict,
    }

def add_to_graph(original_id: str, suspect_id: str,

def action_log(result: dict, action: str, notes: str = ""):

    fps.append({
        "post_id":   result.get("id"),
        "score":     result.get("score_result", {}),
        "logged_at": datetime.datetime.now().isoformat()
    })
    with open(fp_path, "w") as f:
        json.dump(fps, f, indent=2)
    log(f"[FEEDBACK] False positive recorded. Total: {len(fps)}", "WARN")


def get_adaptive_thresholds() -> dict:
    fp_path = "false_positives.json"
    if not os.path.exists(fp_path):
        return {"flag": THRESHOLD_FLAG, "review": THRESHOLD_REVIEW}
    with open(fp_path) as f:
        fps = json.load(f)
    if len(fps) > 10:
        adaptive = min(0.97, THRESHOLD_FLAG + 0.02 * (len(fps) // 10))
        log(f"[FEEDBACK] Adaptive FLAG threshold: {adaptive:.2f} ({len(fps)} false positives)", "WARN")
        return {"flag": adaptive, "review": THRESHOLD_REVIEW}
    return {"flag": THRESHOLD_FLAG, "review": THRESHOLD_REVIEW}


# ── AI Moderation ────────────────────────

@st.cache_resource
def get_llm():
    if not OLLAMA_AVAILABLE:
        return None
    try:
        return ChatOllama(model="qwen2.5:1.5b", temperature=0)
    except Exception:
        return None


def ai_moderate(title: str) -> tuple:
    llm = get_llm()
    if llm is None:
        return "HIGHLIGHT", "AI moderation unavailable — defaulting to scan"
    try:
        prompt = ChatPromptTemplate.from_messages([
            ("system",
             "You are an anti-piracy detection assistant. "
             "Given a Reddit post title, classify it. "
             "Reply: HIGHLIGHT or DISCUSSION, then pipe |, then one sentence reasoning."),
            ("human", "Title: {title}")
        ])
        resp = (prompt | llm).invoke({"title": title}).content.strip()
        if "|" in resp:
            d, r = resp.split("|", 1)
            return d.strip().upper(), r.strip()
        return ("HIGHLIGHT" if "HIGHLIGHT" in resp.upper() else "DISCUSSION"), resp
    except Exception as e:
        log(f"[AI] Moderation error: {e}", "WARN")
        return "HIGHLIGHT", "AI unavailable — defaulting to scan"

# ── Full Pipeline (one post) ─────────────

def run_full_pipeline(post: dict, db: dict, use_ai: bool = True) -> Optional[dict]:
    post_id   = post.get("id", "?")
    title     = post.get("title", "?")
    subreddit = post.get("subreddit", "?")
    permalink = f"https://reddit.com{post.get('permalink','')}"

    if post_id in st.session_state.processed_ids:
        log(f"[PIPELINE] Already processed: {post_id} — skip")
        return None

    if not is_video_post(post):
        log(f"[PIPELINE] Not a video post: {title[:40]}")
        return None

    log(f"\n[PIPELINE] ─── {title[:55]}")

    ai_d, ai_r = "HIGHLIGHT", "AI disabled"
    if use_ai:
        ai_d, ai_r = ai_moderate(title)
        if ai_d == "DISCUSSION":
            log(f"[PIPELINE] AI → DISCUSSION, skipping", "AI")
            return None

    video_path = os.path.join(FRAMES_DIR, f"pipe_{post_id}.mp4")
    ok = ingest_video(permalink, video_path)
    if not ok and post.get("url"):
        ok = ingest_video(post["url"], video_path)
    if not ok:
        log(f"[PIPELINE] Ingestion failed", "ERROR")
        st.session_state.processed_ids.add(post_id)
        return None

    suspect_fp = fingerprint_video(video_path, label=post_id)
    try:
        os.remove(video_path)
    except Exception:
        pass

    best_score = None
    best_asset = None
    thresholds = get_adaptive_thresholds()

    for asset in db["assets"]:
        ref_fp = {
            "label":      asset["name"],
            "phashes":    [h["phash"] for h in asset.get("hashes", [])],
            "pdq_hashes": asset.get("pdq_hashes", []),
            "audio_fp":   asset.get("audio_fp"),
        }
        signals      = detect(suspect_fp, ref_fp)
        score_result = score(signals)

        fs = score_result["final_score"]
        if   fs >= thresholds["flag"]:   score_result["verdict"] = "FLAG"
        elif fs >= thresholds["review"]: score_result["verdict"] = "REVIEW"
        else:                            score_result["verdict"] = "DROP"

        if best_score is None or fs > best_score["final_score"]:
            best_score = score_result
            best_asset = asset["name"]

    if best_score is None:
        log("[PIPELINE] No registered assets to compare against", "WARN")
        return None

    if best_score["verdict"] in ("FLAG", "REVIEW"):
        add_to_graph(original_id=best_asset, suspect_id=post_id,
                     suspect_title=title, final_score=best_score["final_score"],
                     verdict=best_score["verdict"])

    st.session_state.processed_ids.add(post_id)

    result = {
        "id":            post_id,
        "title":         title,
        "subreddit":     subreddit,
        "permalink":     permalink,
        "score":         post.get("score", 0),
        "ai_decision":   ai_d,
        "ai_reason":     ai_r,
        "frames":        suspect_fp["frames"][:4],
        "score_result":  best_score,
        "matched_asset": best_asset,
        "is_violation":  best_score["verdict"] == "FLAG",
        "scanned_at":    datetime.datetime.now().isoformat()
    }

    log(f"[PIPELINE] Done: verdict={best_score['verdict']} score={best_score['final_score']:.3f}",
        "MATCH" if result["is_violation"] else "SUCCESS")
    return result
